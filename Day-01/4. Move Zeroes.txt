    void moveZeroes(vector<int>& nums) {
        int zeroNoInd = 0, nonZeroNoInd = 0, n = nums.size();
        while(nonZeroNoInd < n){
            if(nums[nonZeroNoInd]==0){
                nonZeroNoInd++;
            }else{
                if(nonZeroNoInd==zeroNoInd){
                    nonZeroNoInd++;
                    zeroNoInd++;
                }else{
                    nums[zeroNoInd] = nums[nonZeroNoInd];
                    nums[nonZeroNoInd] = 0;
                    nonZeroNoInd++;
                    zeroNoInd++;
                }
            }
        }
    }
